import { google } from "googleapis"; import fs from "fs"; const { GITHUB_TOKEN, OWNER, REPO, GOOGLE_KEY, FOLDER_ID, INTERVAL = 10000 } = process.env; const auth = new google.auth.GoogleAuth({ credentials: JSON.parse(GOOGLE_KEY), scopes: ["https://www.googleapis.com/auth/drive"], }); const drive = google.drive({ version: "v3", auth }); // ─── Persistent storage ───────────────────────────────────── const ID_FILE = "lastProcessedId.txt"; let lastProcessedId = null; if (fs.existsSync(ID_FILE)) { lastProcessedId = fs.readFileSync(ID_FILE, "utf8").trim(); } function saveLastId(id) { lastProcessedId = id; fs.writeFileSync(ID_FILE, id); } // ─── GitHub Upload & Append Engine ────────────────────────── async function uploadToGitHub(fileName, newText, action) { const path = `generated/${fileName}`; // No forced .html const url = `https://api.github.com/repos/${OWNER}/${REPO}/contents/${path}`; let sha = null; let finalContent = newText; try { // Check if file exists const getRes = await fetch(url, { headers: { Authorization: `token ${GITHUB_TOKEN}`, "User-Agent": "drive-bot" } }); if (getRes.ok) { const fileData = await getRes.json(); sha = fileData.sha; if (action === "append") { // Decode existing content and add new text to the bottom const existingText = Buffer.from(fileData.content, "base64").toString("utf8"); finalContent = existingText + "\n\n" + newText; console.log(`Appending to existing file: ${fileName}`); } else { console.log(`Overwriting existing file: ${fileName}`); } } else if (action === "append") { console.log(`File not found for append, creating new file: ${fileName}`); } const encoded = Buffer.from(finalContent).toString("base64"); const payload = { message: `${action} via drive bot: ${fileName}`, content: encoded, }; if (sha) payload.sha = sha; const putRes = await fetch(url, { method: "PUT", headers: { Authorization: `token ${GITHUB_TOKEN}`, "Content-Type": "application/json", "User-Agent": "drive-bot" }, body: JSON.stringify(payload), }); if (!putRes.ok) { const err = await putRes.text(); throw new Error(`${putRes.status} - ${err}`); } console.log("Success:", path); } catch (err) { console.error("GitHub upload error:", err.message); throw err; // Throw error to prevent Drive rename on failure } } // ─── Main loop ────────────────────────────────────────────── async function main() { try { const res = await drive.files.list({ q: `'${FOLDER_ID}' in parents`, fields: "files(id, name, mimeType, createdTime)", orderBy: "createdTime desc", pageSize: 5, }); // Remove the lastProcessedId check here so it relies entirely on the file prefix const file = res.data.files.find( f => f.mimeType === "application/vnd.google-apps.document" && (f.name.startsWith("upload-") || f.name.startsWith("append-")) ); if (!file) { console.log("No new file"); return; } console.log("Processing:", file.name); let action = "upload"; let cleanName = file.name; // Detect action based on prefix if (file.name.startsWith("append-")) { action = "append"; cleanName = file.name.replace(/^append-/, ""); } else if (file.name.startsWith("upload-")) { action = "upload"; cleanName = file.name.replace(/^upload-/, ""); } // Ensure extension exists if (!cleanName.includes(".")) { cleanName += ".html"; } const contentRes = await drive.files.export({ fileId: file.id, mimeType: "text/plain", }); // ─── Text Cleaner (BOM and Smart Quotes) ─── let cleanText = contentRes.data .replace(/^\uFEFF/, '') .replace(/[\u2018\u2019]/g, "'") .replace(/[\u201C\u201D]/g, '"'); // Wait for GitHub upload to succeed before moving on await uploadToGitHub(cleanName, cleanText, action); // ─── THE LOOP FIX: Rename the file in Drive ─── const newFileName = file.name.replace(/^(upload|append)-/, 'done-'); await drive.files.update({ fileId: file.id, requestBody: { name: newFileName } }); console.log(`Renamed in Drive to: ${newFileName} to prevent loop.`); saveLastId(file.id); } catch (err) { console.error("Main loop error:", err.message); } } setInterval(main, INTERVAL);
